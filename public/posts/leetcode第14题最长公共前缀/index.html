class Solution:
    @staticmethod
    def longestCommonPrefix(strs: list[str]) -> str:
        lcp: str = ''
        max_str: str = max(strs)
        min_str: str = min(strs)
        len_str: int = min(len(max_str), len(min_str))
        for num in range(len_str):
            if max_str[num] == min_str[num]:
                lcp += max_str[num]
            else:
                break
        return lcp
